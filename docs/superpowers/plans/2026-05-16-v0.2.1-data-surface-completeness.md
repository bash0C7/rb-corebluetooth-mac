# v0.2.1 Data-Surface Completeness Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Surface every CoreBluetooth datum the library currently drops. No compat with v0.2.0.

**Architecture:** Same 3-layer stack as v0.2.0 (Ruby → C bridge with `rb_thread_call_without_gvl` → Swift `@c` SE-0495 ABI). Internal JSON wire format moves to `{"ok": true|false, ...}` envelope to carry structured error info uniformly.

**Tech Stack:** Ruby (test-unit / Bundler / mkmf via swift_gem), Swift 6 strict concurrency with `@preconcurrency import CoreBluetooth`, `@unchecked Sendable`, typed `OSAllocatedUnfairLock<State>`, macOS 13+ deployment.

**Source of truth for Apple APIs:** `/Library/Developer/CommandLineTools/SDKs/MacOSX.sdk/System/Library/Frameworks/CoreBluetooth.framework/Versions/A/Headers/` (every CB symbol verified there in spec). Implementers MUST re-verify any signature before writing Swift — no memory.

```swift
struct CBMEnvelope {
    static func ok(_ data: Any) -> String {
        let env: [String: Any] = ["ok": true, "data": data]
        let bytes = try! JSONSerialization.data(withJSONObject: env)
        return String(data: bytes, encoding: .utf8)!
    }
    static func err(_ e: CBMError) -> String {
        let env: [String: Any] = ["ok": false, "error": e.json]
```

```ruby
module CoreBluetoothMac
  class Characteristic
    attr_reader :uuid, :properties, :service, :initial_value

    def initialize(service:, uuid:, properties:, initial_value: nil)
      @service = service
      @uuid = uuid
      @properties = properties.is_a?(Set) ? properties : Set.new(properties)
      @properties.freeze unless @properties.frozen?
      @initial_value = initial_value&.dup&.force_encoding(Encoding::ASCII_8BIT)&.freeze
    end

    def supports?(prop) = @properties.include?(prop)

    def read(timeout: 5.0)
      data = @service.peripheral.__call_native(
        :characteristic_read,
        @service.peripheral.identifier, @service.uuid, @uuid, (timeout * 1000).to_i
      )
      data.force_encoding(Encoding::ASCII_8BIT)
    end

    # write / subscribe / unsubscribe unchanged from v0.2.0
  end
end
```

```ruby
module CoreBluetoothMac
  class Service
    attr_reader :uuid, :peripheral, :is_primary, :included_service_uuids

    def initialize(peripheral:, uuid:, is_primary:)
      @peripheral = peripheral
      @uuid = uuid
      @is_primary = is_primary

      @included_service_uuids = []
    end

    def characteristics
      @characteristics || raise(Error.new("call discover_characteristics first", domain: :discovery))
```

```swift
enum PeripheralEventTag: String { case nameUpdated, servicesInvalidated, disconnected }

struct PeripheralEventState {
    var queue: [(tag: PeripheralEventTag, payload: [String: Any])] = []
    let sem = DispatchSemaphore(value: 0)
}

private let events = OSAllocatedUnfairLock<PeripheralEventState>(initialState: PeripheralEventState())

func pushEvent(tag: PeripheralEventTag, payload: [String: Any]) {
    let sem: DispatchSemaphore = events.withLock { state in
        state.queue.append((tag: tag, payload: payload))
        if state.queue.count > 256 { state.queue.removeFirst() }
        return state.sem
    }
    sem.signal()
}

func pollEvent(timeoutMs: Int32) -> (tag: String, payload: [String: Any])? {
    let immediate: (PeripheralEventTag, [String: Any])? = events.withLock { state in
        state.queue.isEmpty ? nil : state.queue.removeFirst()
    }
    if let e = immediate { return (e.0.rawValue, e.1) }
    let sem = events.withLock { $0.sem }
    let r = sem.wait(timeout: .now() + .milliseconds(Int(timeoutMs)))
    if r == .timedOut { return nil }
    return events.withLock { state in
        guard !state.queue.isEmpty else { return nil }
        let e = state.queue.removeFirst()
        return (e.tag.rawValue, e.payload)
    }
}
```
